<script>

let stars=0;

function addStar(n=1){ stars+=n; document.getElementById("stars").innerText="⭐ "+stars; document.getElementById("soundWin").play(); }

// --- Aprender ---

function startLearn(){

  const area=document.getElementById("gameArea");

  area.innerHTML="<h3>Modo Aprender</h3>";

  for(let t=1;t<=10;t++){

    area.innerHTML+=`<p>Tabla del ${t}: ${Array.from({length:10},(_,i)=>`${t}×${i+1}=${t*(i+1)}`).join(", ")}</p>`;

  }

  addStar(1);

}

// --- Quiz ---

function startQuiz(){

  const area=document.getElementById("gameArea");

  area.innerHTML="<h3>Modo Practicar</h3>";

  let a=Math.floor(Math.random()*10)+1;

  let b=Math.floor(Math.random()*10)+1;

  let correct=a*b;

  area.innerHTML+=`<p>¿Cuánto es ${a}×${b}?</p>`;

  [correct,correct+2,correct-2].sort(()=>Math.random()-0.5).forEach(ans=>{

    area.innerHTML+=`<button class='btn' onclick='checkQuiz(${ans},${correct})'>${ans}</button>`;

  });

}

function checkQuiz(ans,correct){

  if(ans===correct){ document.getElementById("soundCorrect").play(); addStar(1); alert("¡Correcto!"); }

  else { document.getElementById("soundWrong").play(); alert("Ups, intenta otra vez"); }

  startQuiz();

}

// --- Memorama ---

function startMemorama(){

  const area=document.getElementById("gameArea");

  area.innerHTML="<h3>Memorama Multiplicador</h3>";

  let pairs=[ [ "7×8",56 ], [ "6×9",54 ], [ "5×5",25 ], [ "9×9",81 ] ];

  let cards=[];

  pairs.forEach(p=>{ cards.push({val:p[0],match:p[1]}); cards.push({val:p[1],match:p[0]}); });

  cards=cards.sort(()=>Math.random()-0.5);

  let flipped=[], matched=0;

  const grid=document.createElement("div"); grid.className="grid";

  cards.forEach((c,i)=>{

    const tile=document.createElement("div"); tile.className="tile"; tile.innerText="?";

    tile.onclick=function(){

      if(flipped.length<2 && !tile.classList.contains("flipped")){

        tile.classList.add("flipped"); tile.innerText=c.val; flipped.push({tile:tile,val:c.val,match:c.match});

        if(flipped.length===2){

          if(flipped[0].match===flipped[1].val){

            matched++; addStar(1); document.getElementById("soundCorrect").play();

            flipped=[]; if(matched===pairs.length){ alert("¡Ganaste el memorama!"); addStar(3); }

          } else {

            document.getElementById("soundWrong").play();

            setTimeout(()=>{ flipped.forEach(f=>{f.tile.classList.remove("flipped"); f.tile.innerText="?";}); flipped=[]; },1000);

          }

        }

      }

    };

    grid.appendChild(tile);

  });

  area.appendChild(grid);

}
